# Climate Variables
params <- c("palmer", "water_deficit", "soilm", "prcp")

# Area of Interest
AOI <- USAboundaries::us_states() %>%
    filter(state_name == "Iowa")

iowa <- climateR::getTerraClim(
    AOI,
    params,
    startDate = "2012-10-01"
) %>%
    stack() %>%
    setNames(params)

plot(stack(iowa$palmer, iowa$water_deficit, iowa$soilm, iowa$prcp), axes = FALSE, box=FALSE)

indices <- which(!is.na(getValues(iowa)))

kmeans_iowa <- iowa %>%
    getValues() %>%
    na.omit() %>%
    scale() %>%
    kmeans(9)

iowa_raster <- iowa$palmer
iowa_raster[indices] <- kmeans_iowa$cluster

plot(iowa_raster, col = blues9, axes = FALSE, box=FALSE, main = "k-means Clustering with k = 9")
